NICIO ::= <INSTRUCCIONES>  
;
<INSTRUCCIONES> ::= <INSTRUCCIONES>  <INSTRUCCION>  
                | <INSTRUCCION> 
;

<INSTRUCCION> ::= <IMPRECION> 
                | <DECLARACIONV> 
                | <ASIGNACIONV>  
                | <ACTUALIZACIONV> 
                | <IF_P> 
                | <MATCH_P>  
                | <FOR_P>  
                | ><WHILE_P> 
                | <DOWHILE_P>
                | <BREAK_P>
                | <CONINUE_P>
                | <DECLARACION_VECTOR>
                | <ASIG_VECTOR>  
                | ASIG_LISTA  
                | APPEND_LIST 
                | METODOS_FUNCIONES  
                | START_FUNCION  
                | <LLAMADA>  PUNTOYCOMA
                | <STRUCTS> 
                | <INSTANCIACION_STRUCT>  
                | <ASIGNACION_STRUCT>  
                | <RETURN_INS>  
                | error PUNTOYCOMA
                | error PUNTOYCOMA
;

<IMPRECION> ::= PRINTR  PARENTESIS1 <EXPRESION> PARENTESIS2 PUNTOYCOMA 

<EXPRESION> ::= MENOS <EXPRESION> 
            | <EXPRESION> MAS <EXPRESION> 
            | <EXPRESION> MENOS <EXPRESION> 
            | <EXPRESION> POR <EXPRESION> 
            | <EXPRESION> DIVIDIDO <EXPRESION> 
            | <EXPRESION> MODULO <EXPRESION> 
            | <EXPRESION> POR POR <EXPRESION> 
            | PARENTESIS1 TIPO  PARENTESIS2 <EXPRESION> 
            | <EXPRESION> IGUAL IGUAL <EXPRESION> 
            | <EXPRESION> NOT IGUAL <EXPRESION> 
            | <EXPRESION> MENOR <EXPRESION>
            | <EXPRESION> MAYOR <EXPRESION> 
            | <EXPRESION> MENOR IGUAL <EXPRESION> 
            | <EXPRESION> MAYOR IGUAL <EXPRESION> 
            | NOT <EXPRESION> 
            | <EXPRESION> AND <EXPRESION> 
            | <EXPRESION> OR  <EXPRESION> 
            | <EXPRESION> XOR <EXPRESION> 
            | <ACCESO_VECTOR>  
            | <ACCESO_STRUCT>  
            | <REMOVE_LIST> 
            | <LLAMADA>  
            | <ROUND_EXP>  
            | <LENGTH_EXP>  
            | <TOSTRING_EXP>  
            | <FIND_EXP>  
            
;           

<TIPO> ::= INTT 
        | DOUBLET 
        | STRINGT 
        | CHART 
        | BOOLT 

<DECLARACIONV> ::= VARR IDENTIFICADOR  DOSPUNTOS TIPO  IGUAL <EXPRESION>  PUNTOYCOMA 
                | VARR IDENTIFICADOR  DOSPUNTOS TIPO  PUNTOYCOMA 
                | CONSTR IDENTIFICADOR  DOSPUNTOS TIPO  IGUAL <EXPRESION > PUNTOYCOMA 
                | CONSTR IDENTIFICADOR  DOSPUNTOS TIPO  PUNTOYCOMA 
;
    
<ASIGNACIONV> ::= IDENTIFICADOR  IGUAL <EXPRESION> PUNTOYCOMA 
                ;

<ACTUALIZACIONV> ::= IDENTIFICADOR  MAS MAS  
                |   IDENTIFICADOR  MENOS MENOS 
                |   IDENTIFICADOR  MAS MAS PUNTOYCOMA
                |   IDENTIFICADOR  MENOS MENOS PUNTOYCOMA 
                
                ;



<IF_P> ::=  IFR PARENTESIS1 <EXPRESION> PARENTESIS2 LLAVE1 <INSTRUCCIONES> LLAVE2 
        | IFR PARENTESIS1 <EXPRESION> PARENTESIS2 LLAVE1 <INSTRUCCIONES> LLAVE2 ELSER LLAVE1 <INSTRUCCIONES>  LLAVE2 
        | IFR PARENTESIS1 <EXPRESION> PARENTESIS2 LLAVE1 <INSTRUCCIONES> LLAVE2 ELSER <IF_P> 
;


<MATCH_P> ::= MATCH <EXPRESION> LLAVE1 <CASOS>  <CASO_DEFECTO>  LLAVE2 
            | MATCH <EXPRESION> LLAVE1 <CASOS>  LLAVE2 
            | MATCH <EXPRESION> LLAVE1 <CASO_DEFECTO>  

<CASOS> ::= <CASOS>  <CASO>  
        | <CASO>  

<CASO> ::= <EXPRESION> IGUAL MAYOR LLAVE1 <INSTRUCCIONES> LLAVE2

<CASO_DEFECTO> ::= GUIONBAJO IGUAL MAYOR LLAVE1 <INSTRUCCIONES>  LLAVE2 


<FOR_P> ::= FORR PARENTESIS1 <ASIGNACIONV>  <EXPRESION> PUNTOYCOMA <ACT>  PARENTESIS2 LLAVE1 <INSTRUCCIONES>  LLAVE2
<ACT> ::= <ACTUALIZACIONV> 
        | <ASIGNACIONV>:

<WHILE_P> ::= WHILER PARENTESIS1 <EXPRESION> PARENTESIS2 LLAVE1 <INSTRUCCIONES> LLAVE2 

<DOWHILE_P> ::= DOR LLAVE1 <INSTRUCCIONES> LLAVE2 WHILER PARENTESIS1 <EXPRESION> PARENTESIS2 PUNTOYCOMA

<BREAK_P> ::= BREAKR  PUNTOYCOMA 


<CONINUE_P> ::= CONTINUER  PUNTOYCOMA 
//Vectores

<DECLARACION_VECTOR> ::= VARR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 IGUAL CORCHETE1  <LISTA_VALORES>  CORCHETE2 PUNTOYCOMA 
                | CONSTR IDENTIFICADOR  DOSPUNTOS <TIPO>  CORCHETE1 CORCHETE2 IGUAL CORCHETE1 <LISTA_VALORES>  CORCHETE2 PUNTOYCOMA 
                |VARR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 CORCHETE1 CORCHETE2 IGUAL CORCHETE1 <LISTA_VALORES>  CORCHETE2 PUNTOYCOMA 
                |CONSTR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 CORCHETE1 CORCHETE2 IGUAL CORCHETE1 <LISTA_VALORES>  CORCHETE2 PUNTOYCOMA 
                | VARR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 IGUAL IDENTIFICADOR  PUNTOYCOMA
                | CONSTR IDENTIFICADOR DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 IGUAL IDENTIFICADOR  PUNTOYCOMA
                | VARR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 CORCHETE1 CORCHETE2 IGUAL IDENTIFICADOR  PUNTOYCOMA
                | CONSTR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 CORCHETE1 CORCHETE2 IGUAL IDENTIFICADOR  PUNTOYCOMA
                | VARR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 IGUAL <ACCESO_VECTOR>  PUNTOYCOMA
                | CONSTR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 IGUAL <ACCESO_VECTOR>  PUNTOYCOMA
                | VARR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 CORCHETE1 CORCHETE2 IGUAL <ACCESO_VECTOR>  PUNTOYCOMA
                | CONSTR IDENTIFICADOR  DOSPUNTOS <TIPO>   CORCHETE1 CORCHETE2 CORCHETE1 CORCHETE2 IGUAL <ACCESO_VECTOR>  PUNTOYCOMA





<LISTA_VALORES> ::= <LISTA_VALORES>   COMA <EXPRESION>  
                | <LISTA_VALORES>   COMA <LISTA_VALORES2>  
                | <LISTA_VALORES>    
                | <LISTA_VALORES>   
;

<LISTA_VALORES2>   ::= CORCHETE1 <LISTA_VALORES>  CORCHETE2 
;
//Acceso a Vectores
<ACCESO_VECTOR> ::= IDENTIFICADOR  CORCHETE1 <EXPRESION>    CORCHETE2 
                | IDENTIFICADOR  CORCHETE1 <EXPRESION>    CORCHETE2 CORCHETE1 <EXPRESION>    CORCHETE2
<ASIG_VECTOR> ::= IDENTIFICADOR  CORCHETE1 <EXPRESION>    CORCHETE2 CORCHETE1 <EXPRESION>    CORCHETE2 IGUAL <EXPRESION>    PUNTOYCOMA 
                | IDENTIFICADOR  CORCHETE1 <EXPRESION>    CORCHETE2 IGUAL <EXPRESION>    PUNTOYCOMA         
                       

//listas dinamicas
<ASIG_LISTA> ::= LIST <TIPO>  IDENTIFICADOR  IGUAL NEW LIST PARENTESIS1 PARENTESIS2 PUNTOYCOMA
                | LIST <TIPO> IDENTIFICADOR  IGUAL IDENTIFICADOR  

<APPEND_LIST> ::= IDENTIFICADOR  PUNTO APPEND PARENTESIS1 <EXPRESION>    PARENTESIS2 PUNTOYCOMA

<REMOVE_LIST> ::= IDENTIFICADOR  PUNTO REMOVE PARENTESIS1 <EXPRESION>    PARENTESIS2


METODOS_FUNCIONES ::=  <TIPO>  IDENTIFICADOR  PARENTESIS1 PARAMS PARENTESIS2 LLAVE1 <INSTRUCCIONES>  LLAVE2
                        |  <TIPO>  IDENTIFICADOR  PARENTESIS1 PARENTESIS2 LLAVE1 <INSTRUCCIONES>  LLAVE2  
;


PARAMS ::= <PARAMS>  COMA <TIPO>  IDENTIFICADOR    
        | <TIPO>  IDENTIFICADOR                
;

<START_FUNCION> ::=  START IDENTIFICADOR  PARENTESIS1 <PARAMSCALL>  PARENTESIS2 PUNTOYCOMA     
            | START IDENTIFICADOR  PARENTESIS1 PARENTESIS2 PUNTOYCOMA                 
;


<LLAMADA> ::= IDENTIFICADOR  PARENTESIS1 <PARAMSCALL>  PARENTESIS2                
          | IDENTIFICADOR  PARENTESIS1 PARENTESIS2                            
<PARAMSCALL> ::= <PARAMSCALL>  COMA  <EXPRESION>      
            |  <EXPRESION>                         
;


<RETURN_INS> ::= RETURN EXPR <EXPRESION>   PUNTOYCOMA 
            | RETURN  PUNTOYCOMA  
            

//Structs
<STRUCTS> ::= STRUCT LLAVE1 <STRUCTS_LIST>  LLAVE2 IDENTIFICADOR  PUNTOYCOMA 

<STRUCTS_LIST> ::= <STRUCTS_LIST>   IDENTIFICADOR  DOSPUNTOS  <TIPO>  PUNTOYCOMA
                | <STRUCTS_LIST>   IDENTIFICADOR  DOSPUNTOS  IDENTIFICADOR  PUNTOYCOMA
                | IDENTIFICADOR  DOSPUNTOS IDENTIFICADOR  PUNTOYCOMA
                | IDENTIFICADOR  DOSPUNTOS  <TIPO>  PUNTOYCOMA
                ;




<INSTANCIACION_STRUCT> ::= VARR IDENTIFICADOR  DOSPUNTOS IDENTIFICADOR  IGUAL LLAVE1 <VALORES_STRUCT>  LLAVE2 PUNTOYCOMA
                        | CONSTR IDENTIFICADOR  DOSPUNTOS IDENTIFICADOR  IGUAL LLAVE1 <VALORES_STRUCT>  LLAVE2 PUNTOYCOMA
                        
<VALORES_STRUCT> ::= <VALORES_STRUCT>  COMA IDENTIFICADOR  DOSPUNTOS  <EXPRESION>  
                |<VALORES_STRUCT>  COMA IDENTIFICADOR  DOSPUNTOS <INSTANCIACION_STRUCT> 
                | <VALORES_STRUCT>  COMA IDENTIFICADOR  DOSPUNTOS LLAVE1 <VALORES_STRUCT>  LLAVE2
                | IDENTIFICADOR  DOSPUNTOS LLAVE1 <VALORES_STRUCT>  LLAVE2
                | IDENTIFICADOR  DOSPUNTOS <INSTANCIACION_STRUCT> 
                | IDENTIFICADOR  DOSPUNTOS  <EXPRESION>  
                ;

<ACCESO_STRUCT> ::= IDENTIFICADOR  PUNTO IDENTIFICADOR:
                | IDENTIFICADOR  PUNTO IDENTIFICADOR  PUNTO IDENTIFICADOR  

<ASIGNACION_STRUCT> ::= IDENTIFICADOR  PUNTO IDENTIFICADOR  IGUAL  <EXPRESION>   PUNTOYCOMA 
                    | IDENTIFICADOR  PUNTO IDENTIFICADOR  PUNTO IDENTIFICADOR  IGUAL <EXPRESION>  PUNTOYCOMA 
                    ;


// Funciones Nativas 
<ROUND_EXP> ::= ROUND PARENTESIS1 <EXPRESION>  PARENTESIS2
            ;

<LENGTH_EXP> ::= LENGTH PARENTESIS1 <EXPRESION>  PARENTESIS2 

<TOSTRING_EXP> ::= TOSTRING PARENTESIS1 <EXPRESION>  PARENTESIS2

<FIND_EXP> ::= IDENTIFICADOR  PUNTO FIND PARENTESIS1 <EXPRESION>   PARENTESIS2
